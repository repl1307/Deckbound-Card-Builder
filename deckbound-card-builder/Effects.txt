using UnityEngine;

public enum Effects {
    Stun,
    Bleed,
    Shield
}

public enum EffectTargets {
    Self,
    Enemy,
    Ally,
    AllEnemies,
    AllAllies
}

public class EffectDelegates : MonoBehaviour {

    void Start(){

    }

    void Update(){

    }

    // take damage
    public void TakeDamage(Combatant target, int damage){
        target.stats.HP -= damage;
    }
    public void TakeDamage(Combatant[] targets, int damage){
        foreach(Combatant target in targets)
            target.stats.HP -= damage;
    }

    // stun for x turns
    public void Stun(Combatant target, int duration){
        target.effects.Add(Effects.Stun, duration);
    }
    public void Stun(Combatant[] targets, int duration){
        foreach(Combatant target in targets)
            target.effects.Add(Effects.Stun, duration);
    }
}
